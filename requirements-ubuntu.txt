# Installation instructions for Ubuntu 24.04:
# 1. Install system dependencies:
#    sudo apt update
#    sudo apt install -y python3.11 python3.11-venv python3.11-dev
#    sudo apt install -y postgresql postgresql-contrib libpq-dev
#    sudo apt install -y build-essential libssl-dev libffi-dev
#    sudo apt install -y python3-pip
#
# 2. Create virtual environment:
#    python3.11 -m venv venv
#    source venv/bin/activate
#
# 3. Install Python dependencies:
#    pip install --upgrade pip
#    pip install -r requirements-ubuntu.txt
#
# 4. Setup PostgreSQL database:
#    sudo -u postgres createdb cat_database
#    sudo -u postgres createuser cat_user
#    sudo -u postgres psql -c "ALTER USER cat_user PASSWORD 'your_password';"
#    sudo -u postgres psql -c "GRANT ALL PRIVILEGES ON DATABASE cat_database TO cat_user;"
#
# 5. Configure environment variables in .env file:
#    DATABASE_URL=postgresql://cat_user:your_password@localhost/cat_database
#
# 6. Run the application:
#    python server.py
